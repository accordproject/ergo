Definition mk_naked_closure (params:list string) (body:nnrc) : backend_closure :=
  let params := List.map (fun x => (x,None)) params in
  mkClosure
    params
    None
    body.

Definition backend_compose_table (t1 t2:backend_lookup_table) : backend_lookup_table :=
  fun fname =>
    match t1 fname with
    | None => t2 fname
    | Some cl => Some cl
    end.
